**ずんだもん**：「まずはデータの形を決めるのだ！」                                                                                                                                                                                                                                              
   
  using MasterMemory;                                                                                                                                                                                                                                                                         
  using MessagePack;                                                                                                                                                                                                                                                                        
                                                                                                                                                                                                                                                                                              
  [MemoryTable("item"), MessagePackObject(true)]                                                                                                                                                                                                                                              
  public class ItemData                                                                                                                                                                                                                                                                       
  {                                                                                                                                                                                                                                                                                           
      [PrimaryKey]                                                                                                                                                                                                                                                                          
      public int Id { get; set; }

      public string Name { get; set; }                                                                                                                                                                                                                                                        
   
      [SecondaryKey(0)]  // ←カテゴリでも検索できるようにする                                                                                                                                                                                                                                 
      public string Category { get; set; }                                                                                                                                                                                                                                                  
                                                                                                                                                                                                                                                                                              
      public int Price { get; set; }                                                                                                                                                                                                                                                        

      public int Effect { get; set; }
  }

**つむぎ**：「[PrimaryKey]と[SecondaryKey]って何？」                                                                                                                                                                                                                                            
   
**めたん**：「検索用のキーです」                                                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                                            
  - PrimaryKey - メインの検索キー（IDなど）。重複不可。                                                                                                                                                                                                                                       
  - SecondaryKey - サブの検索キー（カテゴリなど）。重複OK。                                                                                                                                                                                                                                 
                                                                                                                                                                                                                                                                                              
**ずんだもん**：「索引みたいなものなのだ！」

  public GachaData DrawSSR()                                                                                                                                                                                                                                                                
  {                                                                                                                                                                                                                                                                                           
      var ssrList = masterDB.GachaTable.FindByRarity("SSR");                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                                              
      // 重みを考慮したランダム抽選                                                                                                                                                                                                                                                           
      int totalWeight = ssrList.Sum(x => x.Weight);
      int random = Random.Range(0, totalWeight);                                                                                                                                                                                                                                              
                                                                                                                                                                                                                                                                                              
      int current = 0;
      foreach (var item in ssrList)                                                                                                                                                                                                                                                           
      {                                                                                                                                                                                                                                                                                     
          current += item.Weight;
          if (random < current)
              return item;                                                                                                                                                                                                                                                                    
      }
                                                                                                                                                                                                                                                                                              
      return ssrList.First();                                                                                                                                                                                                                                                               
  }
